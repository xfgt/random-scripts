IFS - Internal Field Separator;
	Special enviroment variable that determines how the shell recognizes word boundraries when
	processing text strings.
	Bash uses the caharacters defined in IFS to split words during expansion and when reading input
	via the 'read' command.


# 1. Example /etc/passwd
line="root:x:0:0:root:/root:/bin/bash"
# IFS is altered only for the duration of this read command
IFS=":" read -r username password uid gid description home shell <<< "$line"

echo "User $username uses shell $shell"



# 2. Example
OLD_IFS=$IFS

IFS=","

data="Red,Green,Blue"

for color in $data; do
	echo "Color: $color"
done

IFS=$OLD_IFS
